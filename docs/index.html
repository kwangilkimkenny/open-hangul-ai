기본 사용

import { HWPXViewer } from 'open-hangul-ai';
import 'open-hangul-ai/styles';

function App() {
  return (
    <HWPXViewer
      fileUrl="/문서.hwpx"
      enableAI={true}
      onDocumentLoad={(viewer) =>
        console.log('문서 로드 완료!')
      }
    />
  );
}
3

고급 기능

import { HanViewProvider, AIDocumentController } from 'open-hangul-ai';

function AdvancedApp() {
  const aiController = new AIDocumentController({
    provider: 'vertex-ai',
    security: { enableAEGIS: true },
    verification: { enableTruthAnchor: true },
    compliance: ['K-AI-Act', 'EU-AI-Act']
  });

  return (
    <HanViewProvider config={{ aiController }}>
      <HWPXViewer fileUrl="/보고서.hwpx" />
    </HanViewProvider>
  );
}
